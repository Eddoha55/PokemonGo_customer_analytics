# Calculated RFM based on all transactions (summer and fall)
# for each customer (who made at least one transaction)
customer.RFM <- tx.all %>%
  inner_join(customer, by = 'CustomerID') %>%
  group_by(CustomerID) %>%
  summarise(Recency = max(Date), Frequency = n(), Monetary = sum(Value))
